\newcommand{\SmallCapsLabel}[1]{%
  {%
    \fontsize{4.7}{5.2}\selectfont
    \bfseries
    \color{TextMuted}
    \MakeUppercase{#1}%
  }%
}

\newcommand{\StatBox}[4]{%
  % #1 = x-Position links
  % #2 = Breite
  % #3 = Beschriftung
  % #4 = Wert

  \fill[Panel]
    (#1,56.2) rectangle ++(#2,11.4);

  \fill[OverrideGreen]
    (#1,56.2) rectangle ++(0.55,11.4);

  \node[
    anchor=north west,
    text=TextMuted,
    font=\fontsize{4.2}{4.8}\selectfont\bfseries
  ] at ($(#1,67.6)+(1.7,-1.8)$) {%
    \MakeUppercase{#3}%
  };

  \node[
    anchor=west,
    text=TextMain,
    font=\fontsize{13}{13}\selectfont\bfseries
  ] at ($(#1,56.2)+(2.1,4.2)$) {%
    #4%
  };
}

% =============================================================================
% FERTIGKEITSFELD
% =============================================================================

\newcommand{\AbilityBox}[4]{%
  % #1 = untere y-Position
  % #2 = obere y-Position
  % #3 = Slotnummer
  % #4 = Fertigkeits-Shortcut, zum Beispiel \RangeMasterLong

  % Hintergrund und Rahmen
  \fill[Panel]
    (3,#1) rectangle (60.5,#2);

  \draw[
    LineGray,
    line width=0.25mm
  ]
    (3,#1) rectangle (60.5,#2);

  % Gruener Streifen links
  \fill[OverrideGreen]
    (3,#1) rectangle (3.65,#2);

  % Feld fuer die Slotnummer
  \fill[OverrideDarkGreen]
    (5.1,{#2-4.2}) rectangle (9.4,{#2-0.9});

  \node[
    text=OverrideGreen,
    font=\fontsize{5.3}{5.8}\selectfont\bfseries
  ] at (7.25,{#2-2.55}) {%
    #3%
  };

  % Pruefen, ob eine Fertigkeit ausgewaehlt wurde
  \begingroup

  \edef\CurrentAbility{#4}%

  \ifx\CurrentAbility\empty

    % Keine Fertigkeit ausgewaehlt.
    % Rahmen und Slotnummer bleiben trotzdem sichtbar.

  \else

    % Name der Fertigkeit
    \node[
      anchor=north west,
      text=TextMain,
      font=\fontsize{5.0}{5.5}\selectfont\bfseries
    ] at (10.6,{#2-0.9}) {%
      \MakeUppercase{\SPAName{#4}}%
    };

    % Beschreibung der Fertigkeit
    % Lange Texte werden automatisch verkleinert.
% Beschreibung der Fertigkeit
    \node[
      anchor=north west,
      inner sep=0pt,
      text=TextMuted,
      text width=49.4mm,
      align=left,
      font=\fontsize{5.5}{2.2}\selectfont
    ] at (10.6,{#2-4.25}) {%
      \RaggedRight
      \SPAText{#4}%
    };

  \fi

  \endgroup
}
